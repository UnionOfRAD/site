lithium\data\entity\Record

Extends lithium\data\Entity
class

Record class. Represents data such as a row from a database. Records have fields (often known as columns in databases).

class Record extends \lithium\data\Entity {

	protected function _init() {
		parent::_init();
		$this->_handlers += ['stdClass' => function($item) { return $item; }];
	}
}
